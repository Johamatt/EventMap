type Event @model @auth(rules: [{allow: owner, operations: [create, read, update, delete]}, {allow: groups, groups: ["Admin"]}, {allow: public, operations: [read]}]) {
  id: ID!
  infoURL: String
  startingDateTime: String!
  endingDateTime: String!
  source: String!
  location: Location!
  category: [CATEGORY!]
  format: [FORMAT]
  tags: [String]
  company: Company
  eventMedia: [EventMedia]
  Descriptions: Description!
  name: Name!
  external_URL: [AWSURL]
}

type Company @model @auth(rules: [{allow: owner, operations: [create, read, update, delete]}, {allow: groups, groups: ["Admin"]}, {allow: public, operations: [read]}]) {

type Activity @model @auth(rules: [{allow: owner, operations: [create, read, update, delete]}, {allow: groups, groups: ["Admin"]}, {allow: public, operations: [read]}]) {
